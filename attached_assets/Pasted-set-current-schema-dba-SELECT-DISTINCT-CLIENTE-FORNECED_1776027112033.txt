set current schema dba;

SELECT DISTINCT
CLIENTE_FORNECEDOR.IDCLIFOR,
CLIENTE_FORNECEDOR.NOME,
PRODUTO_GRADE.IDSUBPRODUTO,
PRODUTO.DESCRCOMPRODUTO,
PRODUTO_GRADE.SUBDESCRICAO,
COMPRA.DTMOVIMENTO AS DTULTCOMPRA,
COALESCE(COMPRA.QTDPRODUTO, 0) AS QTDULTCOMPRA,
COALESCE(COMPRA.VALUNITBRUTO,0)AS VALULTCOMPRA,
COALESCE(COMPRA.VALUNITBRUTO * COMPRA.QTDENTRADA,0) AS VALORCOMPRA,
COALESCE(COMPRA.VALUNITBRUTO,0.00) AS VALORUNITARIO,
COALESCE(COMPRA.QTDENTRADA, 0.00) AS QTDENTRADA,
COALESCE(VENDA.QTDSAIDA, 0.00) AS QTDSAIDA,
COALESCE( DEVOLUCAOVENDA.QTDEVOLUCAO, 0.00) AS QTDEVOLUCAO,
COALESCE(ATUAL.SALDO, 0) AS SALDOATUAL,
COALESCE(ATUAL.SALDO * COMPRA.VALUNITBRUTO,0) AS VALORATUAL,
COALESCE(RESERVA.RESE, 0) AS QTDRESERVA,
COALESCE(RESERVA_DT.RESE, 0) AS QTDRESERVA_DT,
COALESCE(PC.QTDREPOSICAO, 0) AS QTDREPOSICAO,
COALESCE(SUM( DISTINCT QTDCOMPRAPEN.QTDPENDENTE), 0) AS QTDPENDENTE
FROM
PRODUTO,
PRODUTO_FORNECEDOR LEFT JOIN
PRODUTO_GRADE ON
(PRODUTO_GRADE.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
*
FROM
(
SELECT
EA.IDEMPRESA,
EA.IDPRODUTO,
EA.IDSUBPRODUTO,
EA.QTDPRODUTO,
EA.VALUNITBRUTO,
EA.VALTOTLIQUIDO,
EA.DTMOVIMENTO,
SUM(
CASE WHEN EA.DTMOVIMENTO BETWEEN {d '2026-04-01' } AND {d '2026-04-12' } THEN
EA.QTDPRODUTO
ELSE
0
END) OVER(PARTITION BY EA.IDPRODUTO, EA.IDSUBPRODUTO) AS QTDENTRADA,
ROW_NUMBER() OVER(PARTITION BY EA.IDPRODUTO, EA.IDSUBPRODUTO ORDER BY EA.DTMOVIMENTO DESC, EA.IDPLANILHA DESC) AS RANK
FROM
ESTOQUE_ANALITICO EA,
OPERACAO_INTERNA OI,
PRODUTO_FORNECEDOR
WHERE
EA.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
EA.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
EA.IDOPERACAO = OI.IDOPERACAO AND
OI.TIPOCATEGORIA = 'C' AND
OI.TIPOITEMCATEGORIA = 'C1' AND
OI.IDOPERACAO <> 801 AND
EA.IDEMPRESA IN (3) AND
EA.IDLOCALESTOQUE <> 6
) AS T
WHERE
T.RANK = 1
) AS COMPRA ON
(PRODUTO_GRADE.IDPRODUTO = COMPRA.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = COMPRA.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
*
FROM
(
SELECT
EA.IDEMPRESA,
EA.IDPRODUTO,
EA.IDSUBPRODUTO,
EA.DTMOVIMENTO,
SUM(
CASE WHEN EA.DTMOVIMENTO BETWEEN {d '2026-04-01' } AND {d '2026-04-12' } THEN
EA.QTDPRODUTO
ELSE
0
END) OVER(PARTITION BY EA.IDPRODUTO, EA.IDSUBPRODUTO) AS QTDSAIDA,
ROW_NUMBER() OVER(PARTITION BY EA.IDPRODUTO, EA.IDSUBPRODUTO ORDER BY EA.DTMOVIMENTO DESC, EA.IDPLANILHA DESC) AS RANK
FROM
ESTOQUE_ANALITICO EA,
OPERACAO_INTERNA OI,
PRODUTO_FORNECEDOR
WHERE
EA.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
EA.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
EA.IDOPERACAO = OI.IDOPERACAO AND
OI.TIPOMOVIMENTO = 'V' AND
EA.IDOPERACAO <> 1301 AND
OI.FLAGMOVPRODUTOS = 'T' AND
(EA.NUMSEQUENCIAKIT IS NULL OR EA.NUMSEQUENCIAKIT <= 0) AND
EA.IDEMPRESA IN (3)
) AS T
WHERE
T.RANK = 1
) AS VENDA ON
(PRODUTO_GRADE.IDPRODUTO = VENDA.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = VENDA.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
EA.IDEMPRESA,
EA.IDPRODUTO,
EA.IDSUBPRODUTO,
EA.DTMOVIMENTO,
SUM(
CASE WHEN EA.DTMOVIMENTO BETWEEN {d '2026-04-01' } AND {d '2026-04-12' } THEN
EA.QTDPRODUTO
ELSE
0
END) OVER(PARTITION BY EA.IDPRODUTO, EA.IDSUBPRODUTO) AS QTDEVOLUCAO
FROM
ESTOQUE_ANALITICO EA,
OPERACAO_INTERNA OI,
PRODUTO_FORNECEDOR
WHERE
EA.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
EA.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
EA.IDOPERACAO = OI.IDOPERACAO AND
OI.TIPOCATEGORIA = 'D' AND
OI.TIPOMOVIMENTO = 'E' AND
OI.IDOPERACAO <= 1000 AND
EA.IDOPERACAO <> 1301 AND
OI.FLAGMOVPRODUTOS = 'T' AND
(EA.NUMSEQUENCIAKIT IS NULL OR EA.NUMSEQUENCIAKIT <= 0) AND
EA.IDEMPRESA IN (3)
)AS DEVOLUCAOVENDA ON
(PRODUTO_GRADE.IDPRODUTO = DEVOLUCAOVENDA.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = DEVOLUCAOVENDA.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
ATUAL.IDPRODUTO,
ATUAL.IDSUBPRODUTO,
SUM(ATUAL.QTDATUALESTOQUE) AS SALDO
FROM
ESTOQUE_SALDO_ATUAL ATUAL,
PRODUTO_FORNECEDOR
WHERE
ATUAL.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
ATUAL.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
ATUAL.QTDATUALESTOQUE BETWEEN -999999999.999 AND 999999999.999 AND
ATUAL.IDEMPRESA IN (3)
GROUP BY
ATUAL.IDPRODUTO,
ATUAL.IDSUBPRODUTO
) AS ATUAL ON
(PRODUTO_GRADE.IDPRODUTO = ATUAL.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = ATUAL.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
PRODUTO_COMPRAS.IDPRODUTO,
PRODUTO_COMPRAS.IDSUBPRODUTO,
AVG(PRODUTO_COMPRAS.QTDESTMINIMO) AS QTDREPOSICAO
FROM
PRODUTO_COMPRAS,
PRODUTO_FORNECEDOR
WHERE
PRODUTO_COMPRAS.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
PRODUTO_COMPRAS.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
PRODUTO_COMPRAS.IDEMPRESA IN (3)
GROUP BY
PRODUTO_COMPRAS.IDPRODUTO,
PRODUTO_COMPRAS.IDSUBPRODUTO
) AS PC ON
(PRODUTO_GRADE.IDPRODUTO = PC.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = PC.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
RESERVA.IDPRODUTO,
RESERVA.IDSUBPRODUTO,
SUM(RESERVA.QTDPRODUTO) AS RESE
FROM
ESTOQUE_ANALITICO_TMP RESERVA,
PRODUTO_FORNECEDOR
WHERE
RESERVA.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
RESERVA.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
RESERVA.DTMOVIMENTO BETWEEN {d '2026-04-01' } AND {d '2026-04-12' } AND
RESERVA.QTDPRODUTO BETWEEN -999999999.999 AND 999999999.999 AND
RESERVA.IDEMPRESA IN (3)
GROUP BY
RESERVA.IDPRODUTO,
RESERVA.IDSUBPRODUTO
) AS RESERVA ON
(PRODUTO_GRADE.IDPRODUTO = RESERVA.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = RESERVA.IDSUBPRODUTO)
LEFT JOIN
(
SELECT
RESERVA.IDPRODUTO,
RESERVA.IDSUBPRODUTO,
SUM(RESERVA.QTDPRODUTO) AS RESE
FROM
ESTOQUE_ANALITICO_TMP RESERVA,
PRODUTO_FORNECEDOR
WHERE
RESERVA.IDPRODUTO = PRODUTO_FORNECEDOR.IDPRODUTO AND
RESERVA.IDSUBPRODUTO = PRODUTO_FORNECEDOR.IDSUBPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
RESERVA.QTDPRODUTO BETWEEN -999999999.999 AND 999999999.999 AND
RESERVA.IDEMPRESA IN (3)
GROUP BY
RESERVA.IDPRODUTO,
RESERVA.IDSUBPRODUTO
) AS RESERVA_DT ON
(PRODUTO_GRADE.IDPRODUTO = RESERVA_DT.IDPRODUTO AND
PRODUTO_GRADE.IDSUBPRODUTO = RESERVA_DT.IDSUBPRODUTO)
LEFT JOIN (SELECT
PEDIDO_COMPRA_PROD.IDEMPRESA,
PEDIDO_COMPRA_PROD.IDPRODUTO,
PEDIDO_COMPRA_PROD.IDSUBPRODUTO,
(COALESCE(SUM(PEDIDO_COMPRA_PROD.QTDSOLICITADA),0)-COALESCE(SUM(PEDIDO_COMPRA_PROD.QTDATENDIDA),0)) AS QTDPENDENTE
FROM
PEDIDO_COMPRA_PROD,
PEDIDO_COMPRA
WHERE
PEDIDO_COMPRA_PROD.IDEMPRESA = PEDIDO_COMPRA.IDEMPRESA AND
PEDIDO_COMPRA_PROD.IDPEDIDO = PEDIDO_COMPRA.IDPEDIDO AND
PEDIDO_COMPRA.IDCLIFOR = PEDIDO_COMPRA_PROD.IDCLIFOR AND
UPPER(PEDIDO_COMPRA.FLAGCANCELADO) = 'F' AND
(PEDIDO_COMPRA.DTVALIDADEPEDIDO >= TODAY() OR PEDIDO_COMPRA.DTVALIDADEPEDIDO IS NULL) AND
UPPER(PEDIDO_COMPRA.FLAGPEDIDOBLOQUEADO) = 'F' AND
(PEDIDO_COMPRA.TIPOCATEGORIA IS NULL OR UPPER(PEDIDO_COMPRA.TIPOCATEGORIA) = 'C')AND
PEDIDO_COMPRA_PROD.QTDATENDIDA < PEDIDO_COMPRA_PROD.QTDSOLICITADA AND
PEDIDO_COMPRA_PROD.QTDSOLICITADA > PEDIDO_COMPRA_PROD.QTDATENDIDA AND
PEDIDO_COMPRA_PROD.IDEMPRESA IN (3)
GROUP BY
PEDIDO_COMPRA_PROD.IDEMPRESA,
PEDIDO_COMPRA_PROD.IDPRODUTO,
PEDIDO_COMPRA_PROD.IDSUBPRODUTO
) AS QTDCOMPRAPEN ON
QTDCOMPRAPEN.IDPRODUTO = PRODUTO_GRADE.IDPRODUTO AND
QTDCOMPRAPEN.IDSUBPRODUTO = PRODUTO_GRADE.IDSUBPRODUTO AND
QTDCOMPRAPEN.IDEMPRESA IN (3),
CLIENTE_FORNECEDOR
WHERE
PRODUTO.IDPRODUTO = PRODUTO_GRADE.IDPRODUTO AND
PRODUTO_FORNECEDOR.IDCLIFOR = 1 AND
(UPPER('F') = '%'OR
PRODUTO_GRADE.FLAGINATIVO = UPPER('F')) AND
(UPPER('F') = '%'OR
PRODUTO_GRADE.FLAGINATIVOCOMPRA = UPPER('F')) AND
CLIENTE_FORNECEDOR.IDCLIFOR = PRODUTO_FORNECEDOR.IDCLIFOR
GROUP BY CLIENTE_FORNECEDOR.IDCLIFOR, CLIENTE_FORNECEDOR.NOME, PRODUTO_GRADE.IDSUBPRODUTO, PRODUTO.DESCRCOMPRODUTO, PRODUTO_GRADE.SUBDESCRICAO, COMPRA.DTMOVIMENTO, COALESCE(COMPRA.QTDPRODUTO, 0), COALESCE(COMPRA.VALUNITBRUTO,0), COALESCE(COMPRA.VALUNITBRUTO * COMPRA.QTDENTRADA,0), COALESCE(COMPRA.VALUNITBRUTO,0.00), COALESCE(COMPRA.QTDENTRADA, 0.00), COALESCE(VENDA.QTDSAIDA, 0.00), COALESCE( DEVOLUCAOVENDA.QTDEVOLUCAO, 0.00), COALESCE(ATUAL.SALDO, 0), COALESCE(ATUAL.SALDO * COMPRA.VALUNITBRUTO,0), COALESCE(RESERVA.RESE, 0), COALESCE(RESERVA_DT.RESE, 0), COALESCE(PC.QTDREPOSICAO, 0) FOR READ ONLY
