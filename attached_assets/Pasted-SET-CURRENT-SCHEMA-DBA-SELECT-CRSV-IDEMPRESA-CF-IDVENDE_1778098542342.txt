SET CURRENT SCHEMA DBA;

SELECT
    CRSV.IDEMPRESA,

    CF.IDVENDEDOR AS IDVENDEDOR_CLIENTE,
    VEND.NOME AS NOME_VENDEDOR,

    CF.IDCLIFOR,
    CF.NOME AS NOME_CLIENTE,

    CRSV.IDTITULO,
    CRSV.DIGITOTITULO,
    CRSV.SERIENOTA,

    CASE 
        WHEN COALESCE(N.NUMCUPOMFISCAL, 0) = 0 THEN N.NUMNOTA
        ELSE N.NUMCUPOMFISCAL
    END AS NUMNOTA,

    CRSV.IDPLANILHA,

    CRSV.DTMOVIMENTO,
    CRSV.DTVENCIMENTO,
    CRSV.DTULTIMOPAGAMENTO,

    CASE 
        WHEN CRSV.DTVENCIMENTO < CURRENT DATE THEN
            DAYS(CURRENT DATE) - DAYS(CRSV.DTVENCIMENTO)
        ELSE 0
    END AS DIAS_VENCIDO,

    DECIMAL(COALESCE(CRSV.VALTITULO, 0), 15, 2) AS VALOR_TITULO,
    DECIMAL(COALESCE(CRSV.VALSALDOTITULO, 0), 15, 2) AS VALOR_SALDO_TITULO,
    DECIMAL(COALESCE(CRSV.VALLIQUIDOTITULO, 0), 15, 2) AS VALOR_LIQUIDO_TITULO,

    DECIMAL(
        COALESCE(CRSV.SUMVALJUROSMORA, 0) +
        COALESCE(CRSV.SUMVALJUROSPOSTERGADO, 0),
        15, 2
    ) AS VALOR_JUROS_PENDENTE,

    DECIMAL(COALESCE(CRSV.SUMVALDESCCONCEDIDOTITULO, 0), 15, 2) AS VALOR_DESCONTO_CONCEDIDO,
    DECIMAL(COALESCE(CRSV.SUMVALPAGAMENTOTITULO, 0), 15, 2) AS VALOR_PAGO,

    FPR.DESCRRECEBIMENTO AS FORMA_RECEBIMENTO,

    CRSV.ORIGEMMOVIMENTO,
    CRSV.OBSTITULO,

    CF.ENDERCobranca AS ENDERECO_COBRANCA,
    CF.BAIRROCOBRANCA AS BAIRRO_COBRANCA,
    CI.DESCRCIDADE AS CIDADE_COBRANCA,
    CI.UF AS UF_COBRANCA

FROM
    CONTAS_RECEBER_SALDOS_VIEW CRSV

JOIN CONTAS_RECEBER CR ON
    CR.IDEMPRESA = CRSV.IDEMPRESA
    AND CR.IDCLIFOR = CRSV.IDCLIFOR
    AND CR.IDTITULO = CRSV.IDTITULO
    AND CR.DIGITOTITULO = CRSV.DIGITOTITULO
    AND CR.SERIENOTA = CRSV.SERIENOTA

JOIN CLIENTE_FORNECEDOR CF ON
    CF.IDCLIFOR = CRSV.IDCLIFOR

LEFT JOIN CLIENTE_FORNECEDOR VEND ON
    VEND.IDCLIFOR = CF.IDVENDEDOR

JOIN FORMA_PAGREC FPR ON
    FPR.IDRECEBIMENTO = CRSV.IDRECEBIMENTO

LEFT JOIN NOTAS N ON
    N.IDEMPRESA = CRSV.IDEMPRESA
    AND N.IDPLANILHA = CRSV.IDPLANILHA

LEFT JOIN CIDADES_IBGE CI ON
    CI.IDCIDADE = CF.IDCIDADECOBRANCA

WHERE
    COALESCE(CRSV.VALTITULO, 0) <> 0

    AND (
        COALESCE(CRSV.VALSALDOTITULO, 0) +
        COALESCE(CRSV.VALLIQUIDOTITULO, 0)
    ) > 0

    AND COALESCE(FPR.FLAGCHEQUE, 'F') <> 'T'

    AND CRSV.DTMOVIMENTO >= DATE('1900-01-01')
    AND CRSV.DTMOVIMENTO <= CURRENT DATE

    AND (
        (
            CRSV.FLAGCARTAOPROPRIO = 'T'
            AND CRSV.ORIGEMMOVIMENTO <> 'CTP'
            AND TRIM(COALESCE(CRSV.OBSTITULO, '')) <> 'Débito Fatura Anterior'
            AND NOT EXISTS (
                SELECT
                    1
                FROM
                    CONTAS_RECEBER CR2
                WHERE
                    CR2.IDPLANILHA IN (
                        SELECT
                            CRB.IDPLANILHA
                        FROM
                            CONTAS_RECEBER_BAIXAS CRB
                        WHERE
                            CRB.IDTITULO = CRSV.IDTITULO
                            AND CRB.IDEMPRESA = CRSV.IDEMPRESA
                            AND CRB.DIGITOTITULO = CRSV.DIGITOTITULO
                            AND CRB.IDCLIFOR = CRSV.IDCLIFOR
                            AND CRB.SERIENOTA = CRSV.SERIENOTA
                    )
                    AND CR2.ORIGEMMOVIMENTO = 'CTP'
            )
        )

        OR (
            CRSV.FLAGCARTAOPROPRIO = 'T'
            AND CRSV.ORIGEMMOVIMENTO = 'CTP'
        )

        OR (
            CRSV.FLAGCARTAOPROPRIO = 'F'
            AND (
                NOT (
                    COALESCE(CRSV.IDADMINISTRADORA, 0) > 0
                    AND CRSV.ORIGEMMOVIMENTO = 'CAI'
                )
                OR CR.FLAGAGRUPADO = 'T'
            )
        )
    )

ORDER BY
    VEND.NOME,
    CF.NOME,
    CRSV.DTVENCIMENTO,
    CRSV.IDTITULO

FOR READ ONLY;
