set current schema dba;

WITH ANTECIPACAO AS (
SELECT
OPN.IDPLANILHAPRENOTA
FROM
ORCAMENTO_PRE_NOTA OPN INNER JOIN ORCAMENTO_PROD_NOTA_ANTECIPADA OPNA ON OPNA.IDEMPRESAORCAMENTO = OPN.IDEMPRESAORCAMENTO AND
OPNA.IDORCAMENTO = OPN.IDORCAMENTO
LEFT JOIN ESTOQUE_ANALITICO EA ON OPN.IDEMPRESAPRENOTA = EA.IDEMPRESA AND
OPN.IDPLANILHAPRENOTA = EA.IDPLANILHA
LEFT JOIN ESTOQUE_ANALITICO EA2 ON OPNA.IDEMPRESANOTA = EA2.IDEMPRESA AND
OPNA.IDPLANILHANOTA = EA2.IDPLANILHA AND
OPNA.NUMSEQUENCIANOTA = EA2.NUMSEQUENCIA
LEFT JOIN OPERACAO_INTERNA OP ON OP.IDOPERACAO = EA.IDOPERACAO
LEFT JOIN OPERACAO_INTERNA OP2 ON OP2.IDOPERACAO = EA2.IDOPERACAO
WHERE
OP.TIPOMOVIMENTO = 'V' AND
OP2.TIPOMOVIMENTO = 'V'
)
SELECT
IDEMPRESA,
NOME,
TOTALVENDA,
LUCRO,
TOTALVENDABRUTA,
LUCROBRUTO,
DEVOLUCOES,
QTDVENDA
FROM
(
SELECT
ESTOQUE.IDEMPRESA AS IDEMPRESA,
ESTOQUE.NOME AS NOME,
CAST(SUM(ESTOQUE.TOTALVENDA) AS NUMERIC(18, 6)) AS TOTALVENDA,
CAST(SUM(ESTOQUE.LUCRO) AS NUMERIC(18, 6)) AS LUCRO,
CAST(SUM(ESTOQUE.VENDABRUTA) AS NUMERIC(18, 6)) AS TOTALVENDABRUTA,
CAST(SUM(ESTOQUE.LUCROBRUTO) AS NUMERIC(18, 6)) AS LUCROBRUTO,
CAST(SUM(ESTOQUE.DEVOLUCOES) AS NUMERIC(18, 6)) AS DEVOLUCOES,
CAST(COUNT(DISTINCT ESTOQUE.IDPLANILHA) AS NUMERIC(18,6)) AS QTDVENDA
FROM (
SELECT
EMPRESA.IDEMPRESA,
LTRIM(RTRIM(dba.STRING(EMPRESA.IDEMPRESA))) || ' - ' || EMPRESA.NOMEFANTASIA AS NOME,
CASE WHEN COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 THEN
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
(ESTOQUE_ANALITICO.VALTOTLIQUIDO) * -1
ELSE
(ESTOQUE_ANALITICO.VALTOTLIQUIDO)
END
ELSE
0
END AS TOTALVENDA,
CASE WHEN COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 AND OPERACAO_INTERNA.TIPOMOVIMENTO <> 'E' THEN
ESTOQUE_ANALITICO.VALTOTLIQUIDO
ELSE
0
END AS VENDABRUTA,
CASE WHEN ESTOQUE_ANALITICO.TIPOBAIXAMESTRE = 'K' THEN
(
SELECT
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' THEN
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO) * -1
ELSE
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO)
END
FROM
ESTOQUE_ANALITICO AS EST
WHERE
EST.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND COALESCE(EST.NUMSEQUENCIAKIT, 0) = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND EST.IDSUBPRODUTO IN(
SELECT
PRODKIT.IDSUBPRODUTO
FROM
PRODUTO_KIT AS PRODKIT
WHERE
PRODKIT.IDPRODUTOKIT = ESTOQUE_ANALITICO.IDPRODUTO
AND PRODKIT.IDSUBPRODUTOKIT = ESTOQUE_ANALITICO.IDSUBPRODUTO
)
)
ELSE
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO) * -1
ELSE
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO)
END
END AS LUCRO,
CASE WHEN ESTOQUE_ANALITICO.TIPOBAIXAMESTRE = 'K' THEN
(
SELECT
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' THEN
0
ELSE
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO)
END
FROM
ESTOQUE_ANALITICO AS EST
WHERE
EST.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND COALESCE(EST.NUMSEQUENCIAKIT, 0) = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND EST.IDSUBPRODUTO IN(
SELECT
PRODKIT.IDSUBPRODUTO
FROM
PRODUTO_KIT AS PRODKIT
WHERE
PRODKIT.IDPRODUTOKIT = ESTOQUE_ANALITICO.IDPRODUTO
AND PRODKIT.IDSUBPRODUTOKIT = ESTOQUE_ANALITICO.IDSUBPRODUTO
)
)
ELSE
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
0
ELSE
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO)
END
END AS LUCROBRUTO,
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' AND COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 THEN
ESTOQUE_ANALITICO.VALTOTLIQUIDO
ELSE
0
END AS DEVOLUCOES,
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
NULL
ELSE
ESTOQUE_ANALITICO.IDPLANILHA
END AS IDPLANILHA
FROM
ESTOQUE_ANALITICO
JOIN NOTAS_ENTRADA_SAIDA ON
(NOTAS_ENTRADA_SAIDA.IDOPERACAO = ESTOQUE_ANALITICO.IDOPERACAO
AND NOTAS_ENTRADA_SAIDA.DTMOVIMENTO = ESTOQUE_ANALITICO.DTMOVIMENTO
AND NOTAS_ENTRADA_SAIDA.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND NOTAS_ENTRADA_SAIDA.IDEMPRESA = ESTOQUE_ANALITICO.IDEMPRESA)
JOIN NOTAS ON
(NOTAS.FLAGNOTACANCEL = 'F'
AND NOTAS.IDEMPRESA = NOTAS_ENTRADA_SAIDA.IDEMPRESA
AND NOTAS.IDPLANILHA = NOTAS_ENTRADA_SAIDA.IDPLANILHA)
JOIN EMPRESA ON
(EMPRESA.IDEMPRESA = NOTAS.IDEMPRESA)
JOIN OPERACAO_INTERNA ON
(OPERACAO_INTERNA.IDOPERACAO = ESTOQUE_ANALITICO.IDOPERACAO),
OPERACAO_INTERNA AS CONFIG
WHERE
CONFIG.IDOPERACAO = 3000
AND OPERACAO_INTERNA.FLAGMOVPRODUTOS = 'T'
AND ESTOQUE_ANALITICO.DTMOVIMENTO BETWEEN {d '2026-04-05' } AND {d '2026-04-11' }
AND COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0
AND ESTOQUE_ANALITICO.IDOPERACAO <> 1301
AND NOTAS_ENTRADA_SAIDA.TIPOITEMCATEGORIA NOT IN('D8')
AND OPERACAO_INTERNA.TIPOMOVIMENTO IN ('V', 'E')
AND ESTOQUE_ANALITICO.IDEMPRESA IN (1,2,3)
AND (
NOTAS.SERIENOTA <> 'CAN'
OR (
NOTAS.SERIENOTA = 'CAN'
AND CONFIG.TIPOMOVIMENTO = 'V'
)
)
AND (
SELECT
COUNT(0)
FROM
NOTAS_DEVOLUCAO
JOIN DEVOLUCAO_LOGISTICA_MOVIMENTO ON
(DEVOLUCAO_LOGISTICA_MOVIMENTO.IDEMPRESA = NOTAS_DEVOLUCAO.IDEMPRESA
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDPLANILHA = NOTAS_DEVOLUCAO.IDPLANILHA
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.NUMSEQUENCIADEV = NOTAS_DEVOLUCAO.NUMSEQUENCIADEVOLUCAO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDPRODUTO = NOTAS_DEVOLUCAO.IDPRODUTO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDSUBPRODUTO = NOTAS_DEVOLUCAO.IDSUBPRODUTO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.FLAGGERARREENTREGA = 'T')
JOIN NOTAS_ENTRADA_SAIDA AS ORIGEM_DEVOLUCAO ON
(ORIGEM_DEVOLUCAO.IDEMPRESA = NOTAS_DEVOLUCAO.IDEMPRESA
AND ORIGEM_DEVOLUCAO.IDPLANILHA = NOTAS_DEVOLUCAO.IDPLANILHADEVOLUCAO)
WHERE
NOTAS_DEVOLUCAO.IDEMPRESA = ESTOQUE_ANALITICO.IDEMPRESA
AND NOTAS_DEVOLUCAO.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND NOTAS_DEVOLUCAO.NUMSEQUENCIADEVOLUCAO = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND NOTAS_DEVOLUCAO.IDPRODUTO = ESTOQUE_ANALITICO.IDPRODUTO
AND NOTAS_DEVOLUCAO.IDSUBPRODUTO = ESTOQUE_ANALITICO.IDSUBPRODUTO
) = 0
AND NOT EXISTS (
SELECT 1
FROM ANTECIPACAO
WHERE
ANTECIPACAO.IDPLANILHAPRENOTA = ESTOQUE_ANALITICO.IDPLANILHA
) AND 2 = 2
GROUP BY
EMPRESA.IDEMPRESA,
EMPRESA.NOMEFANTASIA,
OPERACAO_INTERNA.TIPOMOVIMENTO,
ESTOQUE_ANALITICO.IDPRODUTO,
ESTOQUE_ANALITICO.IDSUBPRODUTO,
ESTOQUE_ANALITICO.TIPOBAIXAMESTRE,
ESTOQUE_ANALITICO.VALTOTLIQUIDO,
ESTOQUE_ANALITICO.VALLUCRO,
ESTOQUE_ANALITICO.QTDPRODUTO,
ESTOQUE_ANALITICO.PERMARGEMLUCRO,
ESTOQUE_ANALITICO.IDPLANILHA,
ESTOQUE_ANALITICO.NUMSEQUENCIA,
ESTOQUE_ANALITICO.NUMSEQUENCIAKIT,
ESTOQUE_ANALITICO.IDPLANILHA
) AS ESTOQUE
GROUP BY
ESTOQUE.IDEMPRESA,
ESTOQUE.NOME
UNION ALL
SELECT
ESTOQUE.IDEMPRESA AS IDEMPRESA,
ESTOQUE.NOME || '-DEV(Reentrega)' AS NOME,
CAST(SUM(ESTOQUE.TOTALVENDA) AS NUMERIC(15, 6)) AS TOTALVENDA,
CAST(SUM(ESTOQUE.LUCRO) AS NUMERIC(15, 6)) AS LUCRO,
CAST(SUM(ESTOQUE.VENDABRUTA) AS NUMERIC(15, 6)) AS TOTALVENDABRUTA,
CAST(SUM(ESTOQUE.LUCROBRUTO) AS NUMERIC(15, 6)) AS LUCROBRUTO,
CAST(0 AS NUMERIC(15, 6)) AS DEVOLUCOES,
COUNT(DISTINCT ESTOQUE.IDPLANILHA) AS QTDVENDA
FROM
(
SELECT
EMPRESA.IDEMPRESA,
LTRIM(RTRIM(dba.STRING(EMPRESA.IDEMPRESA))) || ' - ' || EMPRESA.NOMEFANTASIA AS NOME,
CASE WHEN COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 THEN
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
(ESTOQUE_ANALITICO.VALTOTLIQUIDO) * -1
ELSE
(ESTOQUE_ANALITICO.VALTOTLIQUIDO)
END
ELSE
0
END AS TOTALVENDA,
CASE WHEN COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 AND OPERACAO_INTERNA.TIPOMOVIMENTO <> 'E' THEN
ESTOQUE_ANALITICO.VALTOTLIQUIDO
ELSE
0
END AS VENDABRUTA,
CASE WHEN ESTOQUE_ANALITICO.TIPOBAIXAMESTRE = 'K' THEN
(
SELECT
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' THEN
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO) * -1
ELSE
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO)
END
FROM
ESTOQUE_ANALITICO AS EST
WHERE
COALESCE(EST.NUMSEQUENCIAKIT, 0) = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND EST.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND EST.IDSUBPRODUTO IN (
SELECT
PRODKIT.IDSUBPRODUTO
FROM
PRODUTO_KIT AS PRODKIT
WHERE
PRODKIT.IDPRODUTOKIT = ESTOQUE_ANALITICO.IDPRODUTO
AND PRODKIT.IDSUBPRODUTOKIT = ESTOQUE_ANALITICO.IDSUBPRODUTO
)
)
ELSE
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO) * -1
ELSE
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO)
END
END AS LUCRO,
CASE WHEN ESTOQUE_ANALITICO.TIPOBAIXAMESTRE = 'K' THEN
(
SELECT
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' THEN
0
ELSE
SUM(COALESCE(EST.VALLUCRO, 0) * EST.QTDPRODUTO)
END
FROM
ESTOQUE_ANALITICO AS EST
WHERE
COALESCE(EST.NUMSEQUENCIAKIT, 0) = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND EST.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND EST.IDSUBPRODUTO IN (
SELECT
PRODKIT.IDSUBPRODUTO
FROM
PRODUTO_KIT AS PRODKIT
WHERE
PRODKIT.IDPRODUTOKIT = ESTOQUE_ANALITICO.IDPRODUTO
AND PRODKIT.IDSUBPRODUTOKIT = ESTOQUE_ANALITICO.IDSUBPRODUTO
)
)
ELSE
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
0
ELSE
SUM(COALESCE(ESTOQUE_ANALITICO.VALLUCRO, 0) * ESTOQUE_ANALITICO.QTDPRODUTO)
END
END AS LUCROBRUTO,
CASE WHEN OPERACAO_INTERNA.TIPOMOVIMENTO = 'E' AND COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0 THEN
ESTOQUE_ANALITICO.VALTOTLIQUIDO
ELSE
0
END AS DEVOLUCOES,
CASE OPERACAO_INTERNA.TIPOMOVIMENTO WHEN 'E' THEN
NULL
ELSE
ESTOQUE_ANALITICO.IDPLANILHA
END AS IDPLANILHA
FROM
ESTOQUE_ANALITICO
JOIN NOTAS_ENTRADA_SAIDA ON
(NOTAS_ENTRADA_SAIDA.IDOPERACAO = ESTOQUE_ANALITICO.IDOPERACAO
AND NOTAS_ENTRADA_SAIDA.DTMOVIMENTO = ESTOQUE_ANALITICO.DTMOVIMENTO
AND NOTAS_ENTRADA_SAIDA.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND NOTAS_ENTRADA_SAIDA.IDEMPRESA = ESTOQUE_ANALITICO.IDEMPRESA)
JOIN NOTAS ON
(NOTAS.FLAGNOTACANCEL = 'F'
AND NOTAS.IDEMPRESA = NOTAS_ENTRADA_SAIDA.IDEMPRESA
AND NOTAS.IDPLANILHA = NOTAS_ENTRADA_SAIDA.IDPLANILHA)
JOIN EMPRESA ON
(EMPRESA.IDEMPRESA = NOTAS.IDEMPRESA)
JOIN OPERACAO_INTERNA ON
(OPERACAO_INTERNA.IDOPERACAO = ESTOQUE_ANALITICO.IDOPERACAO)
WHERE
COALESCE(ESTOQUE_ANALITICO.NUMSEQUENCIAKIT, 0) <= 0
AND NOTAS_ENTRADA_SAIDA.TIPOITEMCATEGORIA NOT IN('D8')
AND OPERACAO_INTERNA.FLAGMOVPRODUTOS = 'T'
AND OPERACAO_INTERNA.TIPOMOVIMENTO = 'E'
AND ESTOQUE_ANALITICO.IDEMPRESA IN (1,2,3)
AND ESTOQUE_ANALITICO.IDOPERACAO <> 1301
AND ESTOQUE_ANALITICO.DTMOVIMENTO BETWEEN {d '2026-04-05' } AND {d '2026-04-11' }
AND 4 = 4
AND (
SELECT
COUNT(0)
FROM
NOTAS_DEVOLUCAO
JOIN DEVOLUCAO_LOGISTICA_MOVIMENTO ON
(DEVOLUCAO_LOGISTICA_MOVIMENTO.IDEMPRESA = NOTAS_DEVOLUCAO.IDEMPRESA
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDPLANILHA = NOTAS_DEVOLUCAO.IDPLANILHA
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.NUMSEQUENCIADEV = NOTAS_DEVOLUCAO.NUMSEQUENCIADEVOLUCAO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDPRODUTO = NOTAS_DEVOLUCAO.IDPRODUTO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.IDSUBPRODUTO = NOTAS_DEVOLUCAO.IDSUBPRODUTO
AND DEVOLUCAO_LOGISTICA_MOVIMENTO.FLAGGERARREENTREGA = 'T')
JOIN NOTAS_ENTRADA_SAIDA AS ORIGEM_DEVOLUCAO ON
(ORIGEM_DEVOLUCAO.IDEMPRESA = NOTAS_DEVOLUCAO.IDEMPRESA
AND ORIGEM_DEVOLUCAO.IDPLANILHA = NOTAS_DEVOLUCAO.IDPLANILHADEVOLUCAO)
JOIN OPERACAO_INTERNA AS OPERACAO_INTERNA_DEVOLUCAO ON
(OPERACAO_INTERNA_DEVOLUCAO.IDOPERACAO = ORIGEM_DEVOLUCAO.IDOPERACAO)
WHERE
NOTAS_DEVOLUCAO.IDEMPRESA = ESTOQUE_ANALITICO.IDEMPRESA
AND NOTAS_DEVOLUCAO.IDPLANILHA = ESTOQUE_ANALITICO.IDPLANILHA
AND NOTAS_DEVOLUCAO.NUMSEQUENCIADEVOLUCAO = ESTOQUE_ANALITICO.NUMSEQUENCIA
AND NOTAS_DEVOLUCAO.IDPRODUTO = ESTOQUE_ANALITICO.IDPRODUTO
AND NOTAS_DEVOLUCAO.IDSUBPRODUTO = ESTOQUE_ANALITICO.IDSUBPRODUTO
AND OPERACAO_INTERNA.TIPOMOVIMENTO = 'E'
AND OPERACAO_INTERNA_DEVOLUCAO.TIPOMOVIMENTO = 'V'
AND ORIGEM_DEVOLUCAO.IDOPERACAO = 3001
) > 0
AND 2 = 2
GROUP BY
EMPRESA.IDEMPRESA,
EMPRESA.NOMEFANTASIA,
OPERACAO_INTERNA.TIPOMOVIMENTO,
ESTOQUE_ANALITICO.IDPRODUTO,
ESTOQUE_ANALITICO.IDSUBPRODUTO,
ESTOQUE_ANALITICO.TIPOBAIXAMESTRE,
ESTOQUE_ANALITICO.VALTOTLIQUIDO,
ESTOQUE_ANALITICO.VALLUCRO,
ESTOQUE_ANALITICO.QTDPRODUTO,
ESTOQUE_ANALITICO.PERMARGEMLUCRO,
ESTOQUE_ANALITICO.IDPLANILHA,
ESTOQUE_ANALITICO.NUMSEQUENCIA,
ESTOQUE_ANALITICO.NUMSEQUENCIAKIT,
ESTOQUE_ANALITICO.IDPLANILHA
) AS ESTOQUE
GROUP BY
ESTOQUE.IDEMPRESA,
ESTOQUE.NOME
) AS TODAS FOR READ ONLY
